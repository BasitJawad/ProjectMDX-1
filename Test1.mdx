import React, { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0); // For storing data
  let name = "Basit Jawad"
  return (
    <div style={{ padding: '1rem', border: '1px solid #ddd', borderRadius: '4px' }}>
      <h3>Counter: {count}</h3> // Heading
      <button onClick={() => setCount(count + 1)}>Increase</button> // Incrementing values
      <button onClick={() => setCount(count - 1)}>Decrease</button> // Decrementing values
    </div>
  );
}

export default Counter;
